Capitalise Column Names


option validvarname=upcase label=0;
					

%macro upcase_columns(data);
	%let dsid=%sysfunc(open(&data.));
	%let num=%sysfunc(attrn(&dsid, nvars));
	%put &num;
	data &data.;
		set &data.(
			rename=(
				%do i = 1 %to &num;
					%let var&i=%sysfunc(varname(&dsid, &i));
					&&var&i=%sysfunc(upcase(&&var&i))
				%end;
			)
		);
	%let close=%sysfunc(close(&dsid));
	run;
%mend upcase_columns;

%upcase_columns(<data>);

Duplicates

Overview


proc sql;
	create table duplicates as
	select
		date,
		id,
		count(*) as cnt
	from <libname>.<table>
	group by
		date,
		id
	having count(*) > 1;
quit;

Actual Duplicates


%let data = data;
%let col = col;

proc sql;
	create table duplicates as
	select
		a.*,
		b.cnt
	from &data. as a
	inner join (
		select
			&col.,
			count(*) as cnt
		from &data.
		group by
			&col.
		having count(*) > 1
	) as b
		on b.&col. = a.&col.
	where b.cnt ne .
	order by
		&col.;
quit;

Create Snapshots


%macro create_snapshots(dates);
	proc delete data=work.data_raw; run;
	%local i date;
	%do i = 1 %to %sysfunc(countw(&dates., %str( )));

		%let date = %scan(&dates., &i, %str( ));
		%let year_month = %sysfunc(year(&date.))_%sysfunc(putn(%sysfunc(month(&date.)), z2.));

		proc sql;
			create table data_&year_month. as
			select distinct
				&date. format=date9. as snapshot,
				*
			from <libname>.<table>(
				where=(
					<valid_from> <= &date.
					and <valid_to> >= &date.
				)
			)
			group by
				id
			having <priority_variable> = max(<priority_variable>);
		quit;
	
		proc append base=data_raw data=data_&year_month. force;
		run;
	%end;
%mend create_snapshots;

%create_snapshots(&list_dates.);

Excel Single Worksheet


%macro export_xlsx(data);
	proc export
		data=&data.
		dbms=xlsx
		outfile="&path."
		replace;
	run;
%mend export_xlsx;

%export_xlsx(<data>);

Excel Multiple Worksheets


%macro export_xlsx(data, worksheet);
	proc export
		data=&data.
		outfile="&path."
		dbms=xlsx replace;
		sheet=&worksheet.;
	run;
%mend export_xlsx;

%export_xlsx(<data>, "<worksheet>");
